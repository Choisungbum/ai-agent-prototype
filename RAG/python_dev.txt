docker python 개발 환경 

1. dockerfile
FROM python:3.11

# 컨테이너 작업 디렉토리
WORKDIR /app

# 필수 + 개발 편의 시스템 패키지 (문서 변환 패키지 추가)
RUN apt-get update && apt-get install -y \
    build-essential \
    gcc \
    g++ \
    make \
    curl \
    wget \
    git \
    vim \
    nano \
    unzip \
    ca-certificates \
    libgl1 \
    libglib2.0-0 \
    procps \
    net-tools \
    iputils-ping \
    libre-office \
    fonts-noto \
    fonts-noto-cjk \
    && rm -rf /var/lib/apt/lists/*

# python 기본 설정
# 파일 깔끔
ENV PYTHONDONTWRITEBYTECODE=1
# 로그 즉시 출력
ENV PYTHONUNBUFFERED=1"

2. docker build
docker build -t rag-dev .

3. docker run
docker run -it `
  --gpus all `
  -p 8000:8000 `
  -v "C:\Users\csbti\Desktop\ai-agent-prototype\RAG:/app" `
  --name rag-dev-container `
  rag-dev
